import React, { createContext, useContext, useState, useEffect } from 'react';
import { v4 as uuidv4 } from 'uuid';
import { supabase } from '../utils/supabaseClient';

const FamilyContext = createContext();

export const FamilyProvider = ({ children }) => {
    const [nodes, setNodes] = useState([]);
    const [isLoading, setIsLoading] = useState(true);
    const [isAdmin, setIsAdmin] = useState(false);

    // 1. Ambil data pertama kali dari Supabase
    useEffect(() => {
        const fetchFamilyData = async () => {
            try {
                setIsLoading(true);
                const { data, error } = await supabase
                    .from('family_members')
                    .select('*')
                    .order('created_at', { ascending: true });

                if (error) throw error;
                
                // Konversi data array dari format database ke state React
                const formattedData = data.map(item => ({
                    ...item,
                    birthDate: item.birth_date,
                    deathDate: item.death_date,
                    isSpouse: item.is_spouse,
                }));
                
                setNodes(formattedData);
            } catch (err) {
                console.error("Gagal mengambil data dari database:", err.message);
            } finally {
                setIsLoading(false);
            }
        };

        fetchFamilyData();
    }, []);

    const toggleAdmin = () => {
        if (isAdmin) {
            setIsAdmin(false);
        } else {
            const pass = window.prompt("Masukkan Password Admin:");
            if (pass === 'admin123') {
                setIsAdmin(true);
            } else if (pass !== null) {
                alert("Password Salah!");
            }
        }
    };

    // 2. Tambah Anggota Baru ke Supabase
    const addNode = async (newNode) => {
        const id = uuidv4();
        const dbNode = {
            id,
            name: newNode.name,
            birth_date: newNode.birthDate || null,
            death_date: newNode.deathDate || null,
            gender: newNode.gender,
            photo: newNode.photo || null,
            is_spouse: newNode.isSpouse || false,
            spouses: newNode.spouses || [],
            children: newNode.children || [],
            parents: newNode.parents || []
        };

        try {
            const { error } = await supabase.from('family_members').insert([dbNode]);
            if (error) throw error;

            setNodes((prev) => [...prev, { ...newNode, id }]);
        } catch (err) {
            alert("Gagal menambahkan data ke database: " + err.message);
        }
    };

    // 3. Update Data Anggota di Supabase
    const updateNode = async (id, updates) => {
        const dbUpdates = {
            name: updates.name,
            birth_date: updates.birthDate || null,
            death_date: updates.deathDate || null,
            gender: updates.gender,
            photo: updates.photo || null,
        };

        try {
            const { error } = await supabase
                .from('family_members')
                .update(dbUpdates)
                .eq('id', id);

            if (error) throw error;

            setNodes((prev) => prev.map(n => n.id === id ? { ...n, ...updates } : n));
        } catch (err) {
            alert("Gagal memperbarui data di database: " + err.message);
        }
    };

    // 4. Hapus Anggota di Supabase
    const deleteNode = async (id) => {
        if (!isAdmin) {
            alert("Hanya Admin yang bisa menghapus data!");
            return;
        }

        const findDescendants = (nodeId, allNodes, acc = new Set()) => {
            const node = allNodes.find(n => n.id === nodeId);
            if (!node || !node.children) return acc;
            node.children.forEach(cId => {
                if (!acc.has(cId)) {
                    acc.add(cId);
                    findDescendants(cId, allNodes, acc);
                }
            });
            return acc;
        };

        const descendants = findDescendants(id, nodes);
        const affectedBloodline = new Set([id, ...Array.from(descendants)]);
        const targetsToDelete = new Set(Array.from(affectedBloodline));

        nodes.forEach(n => {
            if (affectedBloodline.has(n.id) && n.spouses) {
                n.spouses.forEach(sId => {
                    const spouseNode = nodes.find(sn => sn.id === sId);
                    if (spouseNode && spouseNode.isSpouse) targetsToDelete.add(sId);
                });
            }
        });

        const confirmMsg = targetsToDelete.size > 1
            ? `Keluarga didalam garis ini (${targetsToDelete.size} anggota) akan terhapus semua, apakah Anda yakin ingin menghapusnya?`
            : "Apakah Anda yakin ingin menghapus data ini?";

        if (!window.confirm(confirmMsg)) return;

        const targetsArray = Array.from(targetsToDelete);

        try {
            // Hapus semua node target dari Supabase
            const { error: deleteError } = await supabase
                .from('family_members')
                .delete()
                .in('id', targetsArray);

            if (deleteError) throw deleteError;

            // Bersihkan sisa relasi pada anggota yang masih bertahan
            const updatedNodes = nodes
                .filter(n => !targetsToDelete.has(n.id))
                .map(n => ({
                    ...n,
                    spouses: n.spouses ? n.spouses.filter(sId => !targetsToDelete.has(sId)) : [],
                    children: n.children ? n.children.filter(cId => !targetsToDelete.has(cId)) : [],
                    parents: n.parents ? n.parents.filter(pId => !targetsToDelete.has(pId)) : []
                }));

            // Perbarui relasi sisa anggota di database Supabase
            for (const node of updatedNodes) {
                await supabase
                    .from('family_members')
                    .update({
                        spouses: node.spouses,
                        children: node.children,
                        parents: node.parents
                    })
                    .eq('id', node.id);
            }

            setNodes(updatedNodes);
        } catch (err) {
            alert("Gagal menghapus data dari database: " + err.message);
        }
    };

    // 5. Tambah Pasangan di Supabase
    const addSpouse = async (targetId, spouseData) => {
        const spouseId = uuidv4();
        const targetNode = nodes.find(n => n.id === targetId);
        if (!targetNode) return;

        const updatedSpousesList = [...(targetNode.spouses || []), spouseId];

        const newSpouseDb = {
            id: spouseId,
            name: spouseData.name,
            birth_date: spouseData.birthDate || null,
            death_date: spouseData.deathDate || null,
            gender: spouseData.gender,
            photo: spouseData.photo || null,
            is_spouse: true,
            spouses: [targetId],
            children: [],
            parents: []
        };

        try {
            // 1. Simpan pasangan baru ke Supabase
            const { error: insertError } = await supabase.from('family_members').insert([newSpouseDb]);
            if (insertError) throw insertError;

            // 2. Perbarui kolom spouses milik target di Supabase
            const { error: updateError } = await supabase
                .from('family_members')
                .update({ spouses: updatedSpousesList })
                .eq('id', targetId);

            if (updateError) throw updateError;

            // Perbarui state lokal
            setNodes(prev => [
                ...prev.map(n => n.id === targetId ? { ...n, spouses: updatedSpousesList } : n),
                { ...spouseData, id: spouseId, spouses: [targetId], children: [], isSpouse: true }
            ]);
        } catch (err) {
            alert("Gagal menambahkan pasangan ke database: " + err.message);
        }
    };

    // 6. Tambah Anak di Supabase
    const addChild = async (parentIds, childData) => {
        const parents = Array.isArray(parentIds) ? parentIds : [parentIds];
        const childId = uuidv4();

        const newChildDb = {
            id: childId,
            name: childData.name,
            birth_date: childData.birthDate || null,
            death_date: childData.deathDate || null,
            gender: childData.gender,
            photo: childData.photo || null,
            is_spouse: false,
            spouses: [],
            children: [],
            parents: parents
        };

        try {
            // 1. Simpan anak baru ke Supabase
            const { error: insertError } = await supabase.from('family_members').insert([newChildDb]);
            if (insertError) throw insertError;

            // 2. Perbarui kolom children milik setiap orang tua di Supabase
            for (const parentId of parents) {
                const parentNode = nodes.find(n => n.id === parentId);
                const updatedChildrenList = [...(parentNode?.children || []), childId];

                const { error: updateError } = await supabase
                    .from('family_members')
                    .update({ children: updatedChildrenList })
                    .eq('id', parentId);

                if (updateError) throw updateError;
            }

            // Perbarui state lokal
            setNodes(prev => {
                const updatedNodes = prev.map(n => {
                    if (parents.includes(n.id)) return { ...n, children: [...(n.children || []), childId] };
                    return n;
                });
                return [...updatedNodes, { ...childData, id: childId, parents: parents, spouses: [], children: [] }];
            });
        } catch (err) {
            alert("Gagal menambahkan anak ke database: " + err.message);
        }
    };

    const resetTree = async () => {
        if (!isAdmin) {
            alert("Hanya Admin yang bisa me-reset data!");
            return;
        }
        if (!window.confirm('Apakah Anda yakin ingin me-reset pohon keluarga ke data awal? Semua perubahan akan hilang.')) return;

        try {
            // Hapus seluruh isi tabel di Supabase
            const { error: deleteError } = await supabase
                .from('family_members')
                .delete()
                .neq('id', '00000000-0000-0000-0000-000000000000'); // Trik hapus semua data

            if (deleteError) throw deleteError;

            // Masukkan kembali data inisiasi awal ke Supabase
            const INITIAL_DATA = [
                {
                    id: 'root-1',
                    name: 'Grandfather',
                    birth_date: '1950-01-01',
                    gender: 'male',
                    spouses: ['root-2'],
                    children: ['child-1'],
                },
                {
                    id: 'root-2',
                    name: 'Grandmother',
                    birth_date: '1955-05-15',
                    gender: 'female',
                    spouses: ['root-1'],
                    children: ['child-1'],
                    is_spouse: true,
                },
                {
                    id: 'child-1',
                    name: 'Father',
                    birth_date: '1980-08-20',
                    gender: 'male',
                    parents: ['root-1', 'root-2'],
                }
            ];

            const { error: insertError } = await supabase.from('family_members').insert(INITIAL_DATA);
            if (insertError) throw insertError;

            // Muat ulang halaman untuk memperbarui tampilan
            window.location.reload();
        } catch (err) {
            alert("Gagal melakukan reset database: " + err.message);
        }
    };

    return (
        <FamilyContext.Provider value={{ nodes, addNode, updateNode, deleteNode, addSpouse, addChild, resetTree, isAdmin, toggleAdmin, isLoading }}>
            {children}
        </FamilyContext.Provider>
    );
};

export const useFamily = () => useContext(FamilyContext);
